## 从 base 进行安装

```bash

# 创建虚拟环境
conda create -n object-rel-nav
conda activate object-rel-nav

conda install python=3.9 mamba

# 在 base 环境中使用 mamba 安装 conda 包
conda activate base

# 通用基础包, 主要来自 conda-forge
mamba install -n object-rel-nav \
  pip numpy matplotlib \
  opencv=4.6 cmake=3.14.0 numba=0.57 \
  pyyaml ipykernel networkx h5py natsort \
  transformers einops scikit-learn \
  pgmpy python-igraph pyvis \
  tyro scikit-image spatialmath-python gdown utm seaborn wandb kaggle yacs \
  -c conda-forge --override-channels

# 特殊渠道包
mamba install -n object-rel-nav habitat-sim withbullet \
  -c aihabitat -c conda-forge --override-channels

# GPU/CUDA 依赖包
mamba install -n object-rel-nav faiss-gpu \
  -c conda-forge -c nvidia --override-channels

# 导出 mamba 阶段的环境
mamba env export > object-rel-nav.yml

# 在虚拟环境中安装最后和 torch / pip 相关的包
conda activate object-rel-nav

# 从本地 wheel 仓库安装 torch 三件套及其依赖
pip install torch==claude.1+cu128 torchvision==0.22.1+cu128 torchaudio==claude.1+cu128 \
  -f ~/data/wheels/torch271_cu128_py39/ \
  --no-index

# 在 pip 中记录 mamba 安装的包
python -m pip list --format=freeze > constraints.txt

# 一处维护需要额外 pip 安装的包列表
PIP_EXTRA_PKGS=(
  kornia
  open-clip-torch
  ultralytics
  warmup-scheduler
  efficientnet-pytorch
  vit-pytorch
  lmdb
  prettytable
  diffusers
)

# 先 dry-run, 确认不会意外升级已有包
python -m pip install --dry-run -c constraints.txt "${PIP_EXTRA_PKGS[@]}"

# dry-run 没问题后再正式安装
python -m pip install -c constraints.txt "${PIP_EXTRA_PKGS[@]}"

python -m pip check

# 导出 pip 阶段后的环境
mamba env export > object-rel-nav.after-pip.yml
```
