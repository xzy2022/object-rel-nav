## 本地配置环境

```bash

# 创建虚拟环境
conda create -n object-rel-nav
conda activate object-rel-nav

conda install python=3.9 mamba

# 在 base 环境中使用 mamba 安装 conda 包
conda activate base

# 通用基础包, 主要来自 conda-forge
mamba install -n object-rel-nav \
  pip numpy matplotlib \
  opencv=4.6 cmake=3.14.0 numba=0.57 \
  pyyaml ipykernel networkx h5py natsort \
  transformers einops scikit-learn \
  pgmpy python-igraph pyvis \
  tyro scikit-image spatialmath-python gdown utm seaborn wandb kaggle yacs \
  -c conda-forge --override-channels

# 特殊渠道包
mamba install -n object-rel-nav habitat-sim withbullet \
  -c aihabitat -c conda-forge --override-channels

# GPU/CUDA 依赖包
mamba install -n object-rel-nav faiss-gpu \
  -c conda-forge -c nvidia --override-channels

# 导出 mamba 阶段的环境
mamba env export > object-rel-nav.yml

# 在虚拟环境中安装最后和 torch / pip 相关的包
conda activate object-rel-nav

# 从本地 wheel 仓库安装 torch 三件套及其依赖
pip install torch==2.7.1+cu128 torchvision==0.22.1+cu128 torchaudio==2.7.1+cu128 \
  -f ~/data/wheels/torch271_cu128_py39/ \
  --no-index

# 在 pip 中记录 mamba 安装的包
python -m pip list --format=freeze > constraints.txt

# 一处维护需要额外 pip 安装的包列表
PIP_EXTRA_PKGS=(
  kornia
  open-clip-torch
  ultralytics
  warmup-scheduler
  efficientnet-pytorch
  vit-pytorch
  lmdb
  prettytable
  diffusers
)

# 先 dry-run, 确认不会意外升级已有包
python -m pip install --dry-run -c constraints.txt "${PIP_EXTRA_PKGS[@]}"

# dry-run 没问题后再正式安装
python -m pip install -c constraints.txt "${PIP_EXTRA_PKGS[@]}"

python -m pip check

# 导出 pip 阶段后的环境
mamba env export > object-rel-nav.after-pip.yml
```


## 云服务器配置环境
```bash

conda create -n mamba python=3.9 mamba -c conda-forge
conda activate mamba

mamba env create -f object-rel-nav.yml -n object-rel-nav

# 先安装 torch 然后再固定版本
conda activate object-rel-nav
pip install torch==2.7.1+cu128 torchvision==0.22.1+cu128 torchaudio==2.7.1+cu128 -f https://mirrors.aliyun.com/pytorch-wheels/cu128/

python -m pip list --format=freeze > constraints.txt

# 安装其他 pip 包
PIP_EXTRA_PKGS=(
  kornia
  open-clip-torch
  ultralytics
  warmup-scheduler
  efficientnet-pytorch
  vit-pytorch
  lmdb
  prettytable
  diffusers
)

# 先 dry-run, 确认不会意外升级已有包
python -m pip install --dry-run -c constraints.txt "${PIP_EXTRA_PKGS[@]}"

# dry-run 没问题后再正式安装
python -m pip install -c constraints.txt "${PIP_EXTRA_PKGS[@]}"


```

## 云服务器拉取项目

```bash

# setup git submodule (for objectreact controller)
git clone https://gitee.com/xie-ziyang1/object-rel-nav.git
cd object-rel-nav/
git submodule update --init --recursive

# install habitat-lab
cd libs/
git clone https://gitee.com/xie-ziyang1/habitat-lab.git
cd habitat-lab/
git checkout v0.2.4
pip install -e habitat-lab

```
